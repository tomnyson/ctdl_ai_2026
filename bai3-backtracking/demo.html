def meal_plan(foods, target_kcal):
    result = []
    
    # Sắp xếp mảng tăng dần để Pruning tối ưu
    foods.sort(key=lambda x: x[1])

    def backtrack(start, current_meal, current_kcal):
        # 1. Đạt mục tiêu Kcal
        if current_kcal == target_kcal:
            result.append(current_meal.copy())
            return

        # 2. Duyệt nhánh (CHOOSE, EXPLORE, UNCHOOSE)
        for i in range(start, len(foods)):
            name, kcal = foods[i]
            
            # --- PRUNING TỐI ƯU ---
            if current_kcal + kcal > target_kcal:
                break # Dừng vòng lặp, bỏ qua các món phía sau
            
            current_meal.append(name) # CHOOSE
            
            backtrack(i + 1, current_meal, current_kcal + kcal) # EXPLORE
            
            current_meal.pop() # UNCHOOSE

    backtrack(0, [], 0)
    return result
