Demo thuật toán Quick Sort

Quick Sort chọn một phần tử làm pivot, chia mảng thành hai phần: phần nhỏ hơn pivot và phần lớn hơn pivot, sau đó tiếp tục sắp xếp đệ quy.

Nhấn “Chạy tự động” hoặc “Từng bước” để bắt đầu.
function quickSort(arr, left, right) {
  if (left >= right) return;

  let pivot = arr[right];
  let i = left - 1;

  for (let j = left; j < right; j++) {
    if (arr[j] <= pivot) {
      i++;
      swap(arr, i, j);
    }
  }

  swap(arr, i + 1, right);

  quickSort(arr, left, i);
  quickSort(arr, i + 2, right);
}
