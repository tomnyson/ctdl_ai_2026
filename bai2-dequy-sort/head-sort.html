Demo thuật toán Heap Sort

Heap Sort sắp xếp bằng cách xây dựng max heap. Phần tử lớn nhất nằm ở gốc heap, sau đó được đổi về cuối mảng. Tiếp tục heapify phần còn lại cho đến khi mảng được sắp xếp.

function heapSort(arr) {
  let n = arr.length;

  for (let i = Math.floor(n / 2) - 1; i >= 0; i--) {
    heapify(arr, n, i);
  }

  for (let i = n - 1; i > 0; i--) {
    swap(arr, 0, i);
    heapify(arr, i, 0);
  }
}

function heapify(arr, n, i) {
  let largest = i;
  let left = 2 * i + 1;
  let right = 2 * i + 2;

  if (left < n && arr[left] > arr[largest]) largest = left;
  if (right < n && arr[right] > arr[largest]) largest = right;

  if (largest !== i) {
    swap(arr, i, largest);
    heapify(arr, n, largest);
  }
}
